#include <stdio.h>
#include <stdlib.h>
typedef struct Node {
int data;
struct Node *left;
struct Node *right;
} Node;
Node* createNode(int v) {
Node *n = malloc(sizeof(Node));
n->data = v;
n->left = NULL;
n->right = NULL;
return n;
}
Node* insert(Node *r, int v) {
if (r == NULL) // empty slot
return createNode(v);
if (v < r->data)
r->left = insert(r->left, v);
else if (v > r->data)
r->right = insert(r->right, v);
return r;
}
int main() {
Node *root = NULL;
root = insert(root, 50);
root = insert(root, 30);
root = insert(root, 70);
root = insert(root, 20);
root = insert(root, 40);
root = insert(root, 60);
root = insert(root, 80);
return 0;
}

Node* search(Node *r, int key) {
if (r == NULL) // not found
return NULL;
if (key == r->data) // found!
return r;
if (key < r->data) // go left
return search(r->left, key);
return search(r->right, key);
}
/* Complexity:
Best: O(1) — root match
Avg: O(log n) — balanced
Worst: O(n) — skewed */

C Source — findMin() / findMax()
Node* findMin(Node *r) {
if (r == NULL) return NULL;
while (r->left != NULL)
r = r->left;
return r; // leftmost
}
Node* findMax(Node *r) {
if (r == NULL) return NULL;
while (r->right != NULL)
r = r->right;
return r; // rightmost
}
/* Min: leftmost (no left child) */
/* Max: rightmost (no right child)*/
/* Time: O(h), h = tree height */
int main() {
/* ...build tree... */
Node *mn = findMin(root);
Node *mx = findMax(root);
printf("Min = %d\n", mn->data);
printf("Max = %d\n", mx->data);
return 0;
}

C Source — predecessor()
Node* predecessor(Node *r, int key) {
Node *pred = NULL;
while (r != NULL) {
if (key == r->data) break;
if (key < r->data)
r = r->left;
else {
pred = r; // candidate
r = r->right;
}
}
if (r && r->left) // has left subtree
return findMax(r->left);
return pred;
}
/* Case 1: key has left subtree
→ pred = findMax(left)
Case 2: no left subtree
→ pred = deepest ancestor
where we went right */
int main() {
/* ...build tree... */
Node *pr = predecessor(root, 40);
if (pr)
printf("Predecessor = %d\n", pr->data);
else
printf("No predecessor\n");
return 0;
}

Node* successor(Node *r, int key) {
Node *succ = NULL;
while (r != NULL) {
if (key == r->data) break;
if (key < r->data) {
succ = r; // candidate
r = r->left;
} else
r = r->right;
}
if (r && r->right) // has right subtree
return findMin(r->right);
return succ;
}
/* Case 1: key has right subtree
→ succ = findMin(right)
Case 2: no right subtree
→ succ = deepest ancestor
where we went left */
int main() {
/* ...build tree... */
Node *sc = successor(root, 40);
if (sc)
printf("Successor = %d\n", sc->data);
else
printf("No successor\n");
return 0;
}

Node* deleteNode(Node *r, int key) {
if (r == NULL) return NULL;
if (key < r->data)
r->left = deleteNode(r->left, key);
else if (key > r->data)
r->right = deleteNode(r->right, key);
else { // found!
/* Case 1: leaf */
if (!r->left && !r->right) {
free(r); return NULL;
}
/* Case 2a: no left child */
if (!r->left) {
Node *t=r->right; free(r); return t;
}
/* Case 2b: no right child */
if (!r->right) {
Node *t=r->left; free(r); return t;
}
/* Case 3: two children */
Node *s = findMin(r->right);
r->data = s->data; // copy succ
r->right = deleteNode(r->right, s->data);
}
return r;
}
int main() {
/* ...build tree... */
root = deleteNode(root, 20);
printf("Node %d deleted\n", 20);
return 0;
}
