int isOperator(char c) {
return c=='+'||c=='-'||c=='*'||c=='/'||c=='^';
}
int precedence(char op) {
if (op=='^') return 3;
if (op=='*'||op=='/') return 2;
if (op=='+'||op=='-') return 1;
return 0;
}
void infixToPostfix(char *expr, char *result) {
Stack s; init(&s);
int i = 0, k = 0;
while (expr[i] != '\0') {
char ch = expr[i];
if (isalnum(ch)) {
result[k++] = ch; // operand → output
} else if (ch == '(') {
push(&s, ch); // push '('
} else if (ch == ')') {
while (top(&s) != '(')
result[k++] = pop(&s); // pop until '('
pop(&s); // discard '('
} else if (isOperator(ch)) {
while (!isEmpty(&s) &&
precedence(top(&s)) >= precedence(ch))
result[k++] = pop(&s);
push(&s, ch);
}
i++;
}
while (!isEmpty(&s))
result[k++] = pop(&s); // flush remaining
result[k] = '\0';
}
int main() {
char infix[] = "(a+b)*(c-d)";
char postfix[100];
infixToPostfix(infix, postfix);
printf("Postfix: %s\n", postfix);
return 0;
}

double evalPostfix(char *tokens[], int n) {
Stack s; init(&s);
for (int i = 0; i < n; i++) {
if (isNumber(tokens[i])) {
push(&s, atof(tokens[i])); // number → push directly
} else { // operator
double b = pop(&s), a = pop(&s);
push(&s, apply(a, b, tokens[i][0]));
}
}
return pop(&s);
}

Node* buildTree(char *postfix) {
Stack s; init(&s);
int i = 0;
while (postfix[i] != '\0') {
char ch = postfix[i];
Node *node = newNode(ch);
if (isOperand(ch)) {
push(&s, node); // leaf node
} else { // operator
node->right = pop(&s);
node->left = pop(&s);
push(&s, node);
}
i++;
}
return pop(&s); // root
}

void inorder(Node *r) {
if (r == NULL) return;
if (isOperator(r->data)) printf("(");
inorder(r->left);
printf("%c ", r->data);
inorder(r->right);
if (isOperator(r->data)) printf(")");
}
void preorder(Node *r) {
if (r == NULL) return;
printf("%c ", r->data);
preorder(r->left);
preorder(r->right);
}
void postorder(Node *r) {
if (r == NULL) return;
postorder(r->left);
postorder(r->right);
printf("%c ", r->data);
}
